# CHEM2 Cheat Sheet

$q = mC_s\Delta T$

$q = C_{\mathrm{cal}}\Delta T$

$q = n\Delta H$

$w = -P\Delta V$

$\Delta H^\circ_{\mathrm{rxn}} = \sum n_p\Delta H^\circ_f(\mathrm{products}) - \sum n_r\Delta H^\circ_f(\mathrm{reactants})$

$\Delta H_{\mathrm{rxn}} = \sum \Delta H(\mathrm{Bonds\ Broken}) - \sum \Delta H(\mathrm{Bonds\ Formed})$

$\Delta S^\circ_{\mathrm{rxn}} = \sum n_pS^\circ(\mathrm{products}) - \sum n_rS^\circ(\mathrm{reactants})$

$\Delta G^\circ_{\mathrm{rxn}} = \sum n_p\Delta G^\circ_f(\mathrm{products}) - \sum n_r\Delta G^\circ_f(\mathrm{reactants})$

$\Delta G = \Delta H - T\Delta S$

$\Delta U = q + w$

$\Delta H = \Delta U + P\Delta V$

$\Delta S_{\mathrm{surr}} = \frac{-\Delta H_{\mathrm{sys}}}{T}$

$\Delta G_{\mathrm{rxn}} = \Delta G^\circ_{\mathrm{rxn}} + RT\ln Q$

$\Delta G^\circ_{\mathrm{rxn}} = -RT\ln K$

$S_{\mathrm{gas}} = k_HP_{\mathrm{gas}}$

$P_{\mathrm{solution}} = X_{\mathrm{solvent}}P^\circ_{\mathrm{solvent}}$

$\Delta T_f = mK_fi$

$\Delta T_b = mK_bi$

$\pi = MRTi$

$K_w = [\mathrm{H_3O^+}][\mathrm{OH^-}]$

$K_p = K_c(RT)^{\Delta n}$

$x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}$

$\mathrm{pH} = \mathrm{p}K_a + \log\left(\frac{[\mathrm{Base}]}{[\mathrm{Acid}]}\right)$

$\mathrm{pH} = -\log[\mathrm{H_3O^+}]$

$\Delta G^\circ = -nFE^\circ_{\mathrm{cell}}$

$E^\circ_{\mathrm{cell}} = E^\circ_{\mathrm{cathode}} - E^\circ_{\mathrm{anode}}$

$E^\circ_{\mathrm{cell}} = \frac{0.0592\mathrm{V}}{n}\log K \quad \text{at } 25^\circ\mathrm{C}$

$E_{\mathrm{cell}} = E^\circ_{\mathrm{cell}} - \frac{RT}{nF}\ln Q$

$E_{\mathrm{cell}} = E^\circ_{\mathrm{cell}} - \frac{0.0592\mathrm{V}}{n}\log Q \quad \text{at } 25^\circ\mathrm{C}$

$E^\circ_{\mathrm{cell}} = \frac{RT}{nF}\ln K$

$[\mathrm{A}]_t = -kt + [\mathrm{A}]_0$

$\ln[\mathrm{A}]_t = -kt + \ln[\mathrm{A}]_0$

$\frac{1}{[\mathrm{A}]_t} = kt + \frac{1}{[\mathrm{A}]_0}$

$t_{1/2} = \frac{[\mathrm{A}]_0}{2k}$

$t_{1/2} = \frac{0.693}{k}$

$t_{1/2} = \frac{1}{k[\mathrm{A}]_0}$

$k = Ae^{-E_a/RT}$

# Useful Conversion Factors and Relationships

$1\ \mathrm{lb} = 453.6\ \mathrm{g}$

$1\ \mathrm{in} = 2.54\ \mathrm{cm}\ \text{exactly}$

$1\ \mathrm{mi} = 1.609\ \mathrm{km}$

$1\ \mathrm{km} = 0.6215\ \mathrm{mi}$

$1\ \mathrm{pm} = 1 \times 10^{-12}\ \mathrm{m} = 1 \times 10^{-10}\ \mathrm{cm}$

$1\ \mathrm{atm} = 760\ \mathrm{mmHg} = 760\ \mathrm{torr} = 101{,}325\ \mathrm{N/m^2} = 101{,}325\ \mathrm{Pa}$

$1\ \mathrm{cal} = 4.184\ \mathrm{J}\ \text{exactly}$

$1\ \mathrm{L \cdot atm} = 101.325\ \mathrm{J}$

$1\ \mathrm{J} = 1\ \mathrm{C} \times 1\ \mathrm{V}$

$^\circ\mathrm{C} = (^\circ\mathrm{F} - 32^\circ\mathrm{F}) \times \frac{5^\circ\mathrm{C}}{9^\circ\mathrm{F}}$

$^\circ\mathrm{F} = \frac{9^\circ\mathrm{F}}{5^\circ\mathrm{C}} \times (^\circ\mathrm{C}) + 32^\circ\mathrm{F}$

$\mathrm{K} = (^\circ\mathrm{C} + 273.15^\circ\mathrm{C})\left(\frac{1\ \mathrm{K}}{1^\circ\mathrm{C}}\right)$
